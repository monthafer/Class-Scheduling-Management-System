String search = searchBar.getText().trim();

try {
    Statement stmt = dbconn.createStatement();

    String teacherQuery = "SELECT * FROM teacher WHERE first_name LIKE '%" + search + "%' OR last_name LIKE '%" + search + "%'";
    String studentQuery = "SELECT * FROM student WHERE first_name LIKE '%" + search + "%' OR last_name LIKE '%" + search + "%'";
    String adminQuery = "SELECT * FROM admin WHERE first_name LIKE '%" + search + "%' OR last_name LIKE '%" + search + "%'";

    ResultSet rs = stmt.executeQuery(teacherQuery);
    while (rs.next()) {
        addNameOnPanel(rs, "Teacher");
    }
    rs.close();

    rs = stmt.executeQuery(studentQuery);
    while (rs.next()) {
        addNameOnPanel(rs, "Student");
    }
    rs.close();

    rs = stmt.executeQuery(adminQuery);
    while (rs.next()) {
        addNameOnPanel(rs, "Administrator");
    }
    rs.close();

    stmt.close();

} catch (SQLException ex) {
    Logger.getLogger(Message_Form.class.getName()).log(Level.SEVERE, null, ex);
}

private void addNameOnPanel(ResultSet rs, String position) throws SQLException {
    String name = rs.getString("first_name") + " " + rs.getString("last_name");
    String firstname = rs.getString("first_name");
    String id1 = null;
    if (position.equals("Administrator")) {
        id1 = rs.getString("admin_id");
    } else {
        id1 = String.valueOf(rs.getInt(position.toLowerCase() + "_id"));
    }

    byte[] imagedata = rs.getBytes("img_avatar");
    ImageIcon format;
    if (imagedata != null) {
        format = new ImageIcon(imagedata);
    } else {
        format = new ImageIcon(getClass().getResource("/csms/database/icon/avatarimg.png"));
    }

    nameOnPanel namepanel = new nameOnPanel();
    namepanel.avatar.setImage(format);
    namepanel.namefull.setText(name);
    namepanel.positionfull.setText(position);
    namepanel.addMouseListener(new java.awt.event.MouseAdapter() {
        public void mouseClicked(java.awt.event.MouseEvent evt) {
            chatArea.clearChatBox();
            chatArea.setTitle(name);
            position = position;
            id = id1;
            if (position.equals("Teacher")) {
                chatHistoryTeacher(firstname, format);
            } else if (position.equals("Student")) {
                chatHistoryStudent(firstname, format);
            } else {
                chatHistoryAdmin(firstname, format);
            }
        }
    });

    namePanel.add(namepanel);
}
