    private void initUpdate(){
        choosepanel.removeAll();
        try {
            if (sort.getSelectedIndex() == -1) {
                Statement stmt = dbconn.createStatement();
                ResultSet rs = stmt.executeQuery("SELECT * FROM student");
                while (rs.next()) {
                    String name = rs.getString("first_name") + " " + rs.getString("last_name");
                    String bio = rs.getString("student_desc");
                    String email = rs.getString("email");
                    String contact = rs.getString("contact_no");
                    String facebook = rs.getString("facebook");
                    String address = rs.getString("address");
                    int sgsid = rs.getInt("sgs_id");
                    String sgs = convertSgs(sgsid);

                    byte[] imagedata = rs.getBytes("img_avatar");
                    ImageIcon format;
                    if (imagedata != null) {
                        format = new ImageIcon(imagedata);
                    } else {
                        format = new ImageIcon(getClass().getResource("/csms/database/icon/avatarimg.png"));
                    }

                    StudentPanel student = new StudentPanel();
                    student.setStudent(name);
                    student.setEmail(email);
                    student.setAddress(address);
                    student.setAvatar(format);
                    student.setSgs(sgs);
                    student.setBio(bio);
                    student.setContact(contact);
                    student.setFacebook(facebook);
                    student.initEvent(this);
                    choosepanel.add(student);
                }
            } else {
                Statement stmt = dbconn.createStatement();
                ResultSet rs = stmt.executeQuery("SELECT * FROM student where sgs_id=" + sort.getSelectedIndex());
                while (rs.next()) {
                    String name = rs.getString("first_name") + " " + rs.getString("last_name");
                    String bio = rs.getString("student_desc");
                    String email = rs.getString("email");
                    String contact = rs.getString("contact_no");
                    String facebook = rs.getString("facebook");
                    String address = rs.getString("address");
                    int sgsid = rs.getInt("sgs_id");
                    String sgs = convertSgs(sgsid);

                    byte[] imagedata = rs.getBytes("img_avatar");
                    ImageIcon format;
                    if (imagedata != null) {
                        format = new ImageIcon(imagedata);
                    } else {
                        format = new ImageIcon(getClass().getResource("/csms/database/icon/avatarimg.png"));
                    }

                    StudentPanel student = new StudentPanel();
                    student.setStudent(name);
                    student.setEmail(email);
                    student.setAddress(address);
                    student.setAvatar(format);
                    student.setSgs(sgs);
                    student.setBio(bio);
                    student.setContact(contact);
                    student.setFacebook(facebook);
                    student.initEvent(this);
                    choosepanel.add(student);
                }
            }
        } catch (SQLException ex) {
            Logger.getLogger(Student_Form.class.getName()).log(Level.SEVERE, null, ex);
        }
        choosepanel.repaint();
        choosepanel.revalidate();
    }
    
    private void choice(){
        sort = new Choice();
        sort.setHint("Sort");
        try {
            Statement st = dbconn.createStatement();
            ResultSet rs = st.executeQuery("SELECT * FROM sgs");
            while (rs.next()){
                String sgsChoice = rs.getString("strand") + " " + String.valueOf(rs.getInt("grade")) + "-" + rs.getString("strand");
                sort.addItem(sgsChoice);
            }
        } catch (SQLException ex) {
            Logger.getLogger(Student_Form.class.getName()).log(Level.SEVERE, null, ex);
        }
        sort.setSelectedIndex(-1);
        sort.setForeground(Color.decode("#7A8C8D"));
        choosepanel.add(sort, "w 60%, h 10");
    }
    
    private String convertSgs(int sgsid) throws SQLException{
        Statement stt = dbconn.createStatement();
        ResultSet rss = stt.executeQuery("SELECT * FROM sgs WHERE sgs_id=" + sgsid);
        rss.next();
        String sgs = rss.getString("strand") + " " + String.valueOf(rss.getInt("grade")) + "-" + rss.getString("strand");
        rss.close();
        stt.close();
        return sgs;
    }
